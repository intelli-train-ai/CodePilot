'use client';
import { useProviderModels } from '@/hooks/useProviderModels';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';

import type { TranslationKey } from '@/i18n';

interface RoleModelSelectorProps {

  providerId?: string;
  model?: string;
  onProviderChange: (providerId: string) => void;
  onModelChange: (model: string) => void;

export function RoleModelSelector({ role, providerId, model, onProviderChange, onModelChange }: RoleModelSelectorProps) {
  const { t } = useTranslation();
  const { providerGroups, modelOptions } = useProviderModels(providerId, model);

  const roleKey = `arena.role.${role}` as TranslationKey;
  const providerPlaceholder = t('arena.provider.default' as TranslationKey);

      <Select value={providerId || ''} onValueChange={onProviderChange}>
        <SelectTrigger className="w-40">
          <SelectValue placeholder={providerPlaceholder} />
        </SelectTrigger>
        <SelectContent>
          {providerGroups.map(g => (
            <SelectItem key={g.provider_id} value={g.provider_id}>

          ))}
        </SelectContent>
      </Select>
      <Select value={model || ''} onValueChange={onModelChange}>
        <SelectTrigger className="w-40">
          <SelectValue placeholder={modelPlaceholder} />
        </SelectTrigger>
        <SelectContent>
          {modelOptions.map(m => (
            <SelectItem key={m.value} value={m.value}>
              {m.label}
            </SelectItem>
          ))}
        </SelectContent>
      </Select>
    </div>
  );
}

```typescript
'use client';
import { useState, useCallback } from 'react';
import { useArenaSSE } from '@/hooks/useArenaSSE';
import { LevelCardList } from './LevelCardList';
import { RunControls } from './RunControls';
import type { ArenaViewState, RunParams, ArenaLevelInfo } from './types';

export function ArenaView() {
  const [viewState, setViewState] = useState<ArenaViewState>({ phase: 'levels' });
  const [selectedLevel, setSelectedLevel] = useState<{ worldId: string; levelId: string; name: string; description: string } | null>(null);
  const [showRunControls, setShowRunControls] = useState(false);
  const arenaSSE = useArenaSSE();

  const handleSelectLevel = useCallback((worldId: string, levelId: string) => {
    // 暂存选中的关卡信息，显示 RunControls
    setSelectedLevel({ worldId, levelId, name: '', description: '' });
    setShowRunControls(true);
  }, []);

  const handleStartRun = useCallback(async (params: RunParams) => {
    setShowRunControls(false);
    setViewState({ phase: 'running', runId: '', levelId: params.levelId });
    await arenaSSE.startRun(params);
    // run_started 事件会通过 hook 更新 runId
  }, [arenaSSE]);

  const handleBack = useCallback(() => {
    setShowRunControls(false);
    setSelectedLevel(null);
    setViewState({ phase: 'levels' });
  }, []);

  // 根据 arenaSSE.status 更新 viewState
  // status === 'completed' || status === 'error' -> phase: 'completed'

  if (showRunControls && selectedLevel) {
    return (
      <div className="flex h-full flex-col p-6">
        <RunControls
          levelId={selectedLevel.levelId}
          levelName={selectedLevel.name}
          levelDescription={selectedLevel.description}
          worldId={selectedLevel.worldId}
          onStart={handleStartRun}
          onBack={handleBack}
        />
      </div>
    );
  }

  switch (viewState.phase) {
    case 'levels':
      return (
        <div className="flex h-full flex-col p-6">
          <LevelCardList onSelectLevel={handleSelectLevel} />
        </div>
      );
    case 'running':
    case 'completed':
      return (
        <div className="flex h-full flex-col">
          {/* ConversationStream 将在 Plan 03 中实现 */}
          <div className="flex flex-1 items-center justify-center text-muted-foreground">
            {viewState.phase === 'running' ? 'Running...' : 'Completed'}
          </div>
        </div>
      );
  }
}
```

注意：running/completed 视图的具体内容（ConversationStream + GradeReport）将在 Plan 03 中实现。ArenaView 此时提供完整的状态机骨架和 running/completed 占位。

ArenaView 必须：
- 将 arenaSSE hook 的状态传递给子组件（Plan 03 需要）
- useEffect 监听 arenaSSE.status 变化，当 status 变为 'completed' 或 'error' 时更新 viewState.phase 为 'completed'
- 当 arenaSSE.runId 变化时更新 viewState 中的 runId
