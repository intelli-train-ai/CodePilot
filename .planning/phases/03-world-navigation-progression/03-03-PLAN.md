1. **在 `src/arena/db.ts` 末尾新增 `getRunsByLevel` 函数:**

```typescript
/**
 * 查询指定关卡的所有运行记录（按创建时间倒序）。
 * 包含 grade 通关状态用于历史列表展示。
 */
export function getRunsByLevel(levelId: string): Array<{
  id: string;
  status: ArenaRunStatus;
  passed: number | null;
  turn_count: number;
  token_usage_total: number;
  termination_reason: TerminationReason | null;
  created_at: string;
  completed_at: string | null;
  grade_passed: number | null;
}> {
  const db = getDb();
  return db.prepare(`
    SELECT r.id, r.status, r.passed, r.turn_count, r.token_usage_total,
           r.termination_reason, r.created_at, r.completed_at,
           g.passed as grade_passed
    FROM arena_runs r
    LEFT JOIN arena_grades g ON g.run_id = r.id
    WHERE r.level_id = ?
    ORDER BY r.created_at DESC
  `).all(levelId) as Array<{
    id: string;
    status: ArenaRunStatus;
    passed: number | null;
    turn_count: number;
    token_usage_total: number;
    termination_reason: TerminationReason | null;
    created_at: string;
    completed_at: string | null;
    grade_passed: number | null;
  }>;
}
```

2. **新建 `src/app/api/arena/levels/[levelId]/runs/route.ts`:**

创建目录结构 `src/app/api/arena/levels/[levelId]/runs/`，新建 `route.ts`:

```typescript
import { NextRequest } from 'next/server';
import { requireAuth } from '@/lib/auth';
import { getRunsByLevel } from '@/arena/db';

export const runtime = 'nodejs';
export const dynamic = 'force-dynamic';

/**
 * GET /api/arena/levels/:levelId/runs
 *
 * 返回指定关卡的运行历史。
 * Response: { runs: Array<{ id, status, passed, turnCount, tokenUsage, createdAt, completedAt }> }
 */
export async function GET(
  request: NextRequest,
  { params }: { params: Promise<{ levelId: string }> },
) {
  const authError = requireAuth(request);
  if (authError) return authError;

  try {
    const { levelId } = await params;
    if (!levelId || typeof levelId !== 'string') {
      return new Response(JSON.stringify({ error: 'Missing levelId' }), {
        status: 400,
        headers: { 'Content-Type': 'application/json' },
      });
    }

    const rows = getRunsByLevel(levelId);
    const runs = rows.map(r => ({
      id: r.id,
      status: r.status,
      passed: r.grade_passed === 1,
      turnCount: r.turn_count,
      tokenUsage: r.token_usage_total,
      createdAt: r.created_at,
      completedAt: r.completed_at,
    }));

    return new Response(JSON.stringify({ runs }), {
      status: 200,
      headers: { 'Content-Type': 'application/json' },
    });
  } catch (err) {
    return new Response(JSON.stringify({
      error: err instanceof Error ? err.message : 'Failed to load run history',
    }), {
      status: 500,
      headers: { 'Content-Type': 'application/json' },
    });
  }
}
```

注意: Next.js 15+ App Router 的 params 是 Promise，需要 await。参考项目其他 dynamic route 的写法。
