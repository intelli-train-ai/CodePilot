```typescript
import { NextRequest } from 'next/server';
import { runArenaOrchestration } from '@/arena/engine/orchestrator';
import { formatArenaSSE } from '@/arena/types';

/**
 * POST /api/arena/run
 *
 * 启动一次 Arena 运行。返回 SSE 流实时推送进度。
 *
 * Request body:
 * {
 *   worldId: string;       // 世界 ID
 *   levelId: string;       // 关卡 ID
 *   providerId?: string;   // 默认 provider（可选，关卡配置优先）
 *   model?: string;        // 默认 model（可选，关卡配置优先）
 * }
 *
 * Response: text/event-stream
 * Events: run_started, gatekeeper_message, challenger_delta, challenger_message,
 *         turn_completed, grading_started, grade_result, run_completed, run_error, token_usage
 */
export async function POST(request: NextRequest) {
  // 认证检查（复用项目既有的 requireAuth 模式）
  // 检查 X-Auth-Token header
  const authToken = request.headers.get('x-auth-token');
  const expectedToken = process.env.CODEPILOT_ACCESS_TOKEN;
  if (expectedToken && authToken !== expectedToken) {
    return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 });
  }

  // 解析请求体
  let body: { worldId: string; levelId: string; providerId?: string; model?: string };
  try {
    body = await request.json();
  } catch {
    return new Response(JSON.stringify({ error: 'Invalid JSON body' }), { status: 400 });
  }

  if (!body.worldId || !body.levelId) {
    return new Response(JSON.stringify({ error: 'worldId and levelId are required' }), { status: 400 });
  }

  // 创建 AbortController 监听客户端断开
  const abortController = new AbortController();
  request.signal.addEventListener('abort', () => abortController.abort());

  // 创建 SSE ReadableStream
  const stream = new ReadableStream({
    async start(controller) {
      try {
        const orchestration = runArenaOrchestration({
          worldId: body.worldId,
          levelId: body.levelId,
          defaultProviderId: body.providerId,
          defaultModel: body.model,
          abortSignal: abortController.signal,
        });

        for await (const event of orchestration) {
          try {
            controller.enqueue(formatArenaSSE(event));
          } catch {
            // controller 已关闭（客户端断开），安全退出
            break;
          }
        }
      } catch (err) {
        try {
          controller.enqueue(formatArenaSSE({
            type: 'run_error',
            data: { error: err instanceof Error ? err.message : String(err) },
          }));
        } catch { /* controller 已关闭 */ }
      } finally {
        try { controller.close(); } catch { /* 已关闭 */ }
      }
    },
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
      'X-Accel-Buffering': 'no',
    },
  });
}
```

2. 创建 `src/app/api/arena/levels/route.ts`：

```typescript
import { NextRequest } from 'next/server';
import { loadAllWorlds } from '@/arena/level-loader';

/**
 * GET /api/arena/levels
 *
 * 返回所有可用世界和关卡列表。
 *
 * Response:
 * {
 *   worlds: Array<{
 *     id: string;
 *     name: string;
 *     description: string;
 *     icon?: string;
 *     levels: Array<{
 *       id: string;
 *       name: string;
 *       description: string;
 *       maxTurns: number;
 *       rubricCount: number;
 *     }>
 *   }>
 * }
 */
export async function GET(request: NextRequest) {
  // 认证检查
  const authToken = request.headers.get('x-auth-token');
  const expectedToken = process.env.CODEPILOT_ACCESS_TOKEN;
  if (expectedToken && authToken !== expectedToken) {
    return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 });
  }

  try {
    const worlds = loadAllWorlds();

    const response = {
      worlds: worlds.map(w => ({
        id: w.config.id,
        name: w.config.name,
        description: w.config.description,
        icon: w.config.icon,
        levels: w.levels.map(l => ({
          id: l.config.id,
          name: l.config.name,
          description: l.config.description,
          maxTurns: l.config.maxTurns,
          rubricCount: l.config.rubric.length,
        })),
      })),
    };

    return new Response(JSON.stringify(response), {
      status: 200,
      headers: { 'Content-Type': 'application/json' },
    });
  } catch (err) {
    return new Response(JSON.stringify({
      error: err instanceof Error ? err.message : 'Failed to load levels',
    }), { status: 500 });
  }
}
```

import { describe, it } from 'node:test';
import assert from 'node:assert/strict';
import { loadAllWorlds } from '@/arena/level-loader';

describe('Arena levels API', () => {
  it('loadAllWorlds returns customer-service world', () => {
    const worlds = loadAllWorlds();
    assert.ok(worlds.length >= 1, 'Should have at least 1 world');
    const cs = worlds.find(w => w.config.id === 'customer-service');
    assert.ok(cs, 'customer-service world should exist');
  });

  it('customer-service world has 2 levels in correct order', () => {
    const worlds = loadAllWorlds();

    assert.equal(cs.levels.length, 2);
    assert.equal(cs.levels[0].config.id, 'cs-greeting');
    assert.equal(cs.levels[1].config.id, 'cs-complaint');

  it('level objects have required fields', () => {
    const worlds = loadAllWorlds();
    const cs = worlds.find(w => w.config.id === 'customer-service')!;
    for (const level of cs.levels) {
