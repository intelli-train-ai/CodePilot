```typescript
/**
 * 调用 Gatekeeper LLM，获取结构化输出（含 shouldEnd）。
 * 使用 generateText + Output.object()（非 streamText），因为需要完整解析 shouldEnd 后才能决定是否继续循环。
 *
 * per D-02: 解析失败重试一次，再失败返回 null 表示 parse_failure。
 *
 * @param transcript - 对话历史，转换为 AI SDK messages 格式
 * @param level - 关卡配置（含 gatekeeperSystemPrompt）
 * @param abortSignal - 取消信号
 * @returns GatekeeperCallResult | null（null 表示结构化输出解析两次失败）
 */
export async function callGatekeeper(opts: {
  transcript: Array<{ role: 'user' | 'assistant'; content: string }>;
  level: LevelConfig;
  abortSignal?: AbortSignal;
}): Promise<GatekeeperCallResult | null> {
  // 构建 AI SDK model：使用 level.roleConfig.gatekeeper 的 providerId/model，
  // 如果未配置则 fallback 到全局默认（resolveProvider() 无参调用）
  const providerId = opts.level.roleConfig?.gatekeeper?.providerId || '';
  const modelId = opts.level.roleConfig?.gatekeeper?.model || '';
  // 使用与 text-generator.ts 相同的 model 构建模式：
  // resolveProvider → toAiSdkConfig → createXxx(config) → model(modelId)
  // 需要从 text-generator.ts 提取 buildModel 辅助函数，或在此文件内复制 model 构建逻辑
  // （参考 text-generator.ts 第 49-97 行的 switch-case）

  // 重试逻辑：per D-02，最多尝试 2 次
  for (let attempt = 0; attempt < 2; attempt++) {
    try {
      const { output, usage } = await generateText({
        model: /* 构建的 model 实例 */,
        output: Output.object({ schema: GatekeeperOutputSchema }),
        system: opts.level.gatekeeperSystemPrompt,
        messages: opts.transcript,
        abortSignal: opts.abortSignal || AbortSignal.timeout(60_000),
      });

      if (!output) {
        // Output.object() 返回 null 表示解析失败
        if (attempt === 0) continue; // 重试一次
        return null;
      }

      return {
        output,
        usage: {
          totalTokens: usage.totalTokens,
          promptTokens: usage.promptTokens,
          completionTokens: usage.completionTokens,
        },
      };
    } catch (err) {
      if (attempt === 0) continue; // 重试一次
      console.warn('[arena:gatekeeper] Structured output failed after 2 attempts:', err);
      return null;
    }
  }
  return null;
}
```

```typescript
import { streamTextFromProvider } from '@/lib/text-generator';
import type { LevelConfig } from '../schemas/level-config';

export interface ChallengerCallResult {
  fullContent: string;
  usage: { totalTokens: number; promptTokens: number; completionTokens: number };
}

/**
 * 调用 Challenger LLM，流式生成回复。
 * 使用 streamTextFromProvider（已在 Plan 01 中扩展支持 messages）。
 *
 * 返回一个 async generator，yield 每个 text chunk（用于 SSE delta 推送），
 * 以及最终的完整内容和 usage 统计。
 *
 * @param transcript - 对话历史（Gatekeeper 消息作为 user，Challenger 消息作为 assistant）
 * @param level - 关卡配置
 */
export async function callChallenger(opts: {
  transcript: Array<{ role: 'user' | 'assistant'; content: string }>;
  level: LevelConfig;
  abortSignal?: AbortSignal;
  onDelta?: (delta: string) => void;
}): Promise<ChallengerCallResult> {
  const providerId = opts.level.roleConfig?.challenger?.providerId || '';
  const modelId = opts.level.roleConfig?.challenger?.model || '';

  const chunks: string[] = [];
  // 使用 streamTextFromProvider 的 messages 模式
  for await (const chunk of streamTextFromProvider({
    providerId,
    model: modelId,
    system: opts.level.challengerSystemPrompt,
    prompt: '',  // messages 模式下 prompt 不使用
    messages: opts.transcript,
    abortSignal: opts.abortSignal,
  })) {
    chunks.push(chunk);
    opts.onDelta?.(chunk);
  }

  const fullContent = chunks.join('');
  // 注意：streamTextFromProvider 当前不返回 usage。
  // 需要扩展 streamTextFromProvider 返回 usage，或在这里使用 AI SDK 的
  // result.usage property。实现时需确认最佳方式。
  // 临时方案：通过 text chunk 估算（不精确）。
  // 推荐方案：修改 streamTextFromProvider 使其也能返回 usage（在 generator 结束后可通过 result 对象获取）。

  return {
    fullContent,
    usage: { totalTokens: 0, promptTokens: 0, completionTokens: 0 }, // 需在实现时获取实际 usage
  };
}
```

- 方案 B：在 challenger.ts 中直接调用 AI SDK `streamText`（绕过 text-generator 的 generator wrapper），这样可以直接访问 `result.usage`

推荐方案 A，保持通过 text-generator 的统一调用路径（per INTG-01）。具体实现：在 `streamTextFromProvider` 中让 `result` 对象可访问，或者新增一个 `streamTextWithUsage` 函数返回 `{ stream: AsyncIterable<string>; getUsage: () => Promise<TokenUsage> }`。

```typescript
  const systemPrompt = opts.level.graderSystemPrompt || defaultGraderSystem;

  // 使用 model-builder 构建 Grader model
  const { output, usage } = await generateText({

    throw new Error('[arena:grader] Failed to generate structured grading output');

    output,
    usage: {
      totalTokens: usage.totalTokens,

/**
 * 构建 Grader 的 prompt，包含 rubric 详情和完整对话 transcript。
 */
function buildGraderPrompt(
  transcript: Array<{ role: string; content: string; turn: number }>,
  rubric: RubricItem[],
): string {
  let prompt = '## Grading Rubric\n\n';

  for (const item of rubric) {
    prompt += `### ${item.name} (${item.type})\n`;
    prompt += `${item.description}\n`;
    if (item.type === 'performance' && item.gradeDescriptions) {
      for (const [grade, desc] of Object.entries(item.gradeDescriptions)) {
        prompt += `- **${grade}**: ${desc}\n`;
      }
    }
    prompt += '\n';

  prompt += '## Conversation Transcript\n\n';
  for (const msg of transcript) {
    const roleLabel = msg.role === 'gatekeeper' ? 'Gatekeeper' : 'Challenger';
    prompt += `**Turn ${msg.turn} - ${roleLabel}:**\n${msg.content}\n\n`;
  }

  prompt += '## Instructions\n\n';
  prompt += 'Evaluate the Challenger based on the rubric above. For each required criterion, determine Pass or Fail. For each performance dimension, assign a grade (A/B/C/D) using the grade descriptions as anchors. Provide up to 3 improvement suggestions referencing specific turn numbers.';

  return prompt;
}
```

```typescript
const DEFAULT_TOKEN_BUDGET = 200_000;
const GRADER_RESERVE = 8_000; // 为 Grader 预留的 token 空间

export class TokenTracker {
  private budget: number;
  private used: number = 0;

  constructor(budget?: number) {
    this.budget = budget ?? DEFAULT_TOKEN_BUDGET;
  }

  /** 记录 token 消耗 */
  consume(tokens: number): void {
    this.used += tokens;
  }

  /** 剩余可用 token */
  get remaining(): number {
    return this.budget - this.used;
  }

  /** 已消耗总量 */
  get totalUsed(): number {
    return this.used;
  }

  /** 预算是否已耗尽 */
  get isExhausted(): boolean {
    return this.remaining <= 0;
  }

  /** 是否有足够 token 继续（含 Grader 预留） */
  hasEnoughForNextTurn(): boolean {
    return this.remaining > GRADER_RESERVE;
  }

  /** 是否有足够 token 进行指定操作 */
  hasEnoughFor(estimatedTokens: number): boolean {
    return this.remaining >= estimatedTokens;
  }
}
```

```typescript
 * 循环终止条件（按优先级）：
 * 1. Gatekeeper shouldEnd=true (ORCH-02) → terminationReason='gatekeeper_end'
 * 2. Token budget exhausted (ORCH-04, D-03) → terminationReason='token_budget'
 * 3. maxTurns reached (ORCH-03) → terminationReason='max_turns'
 * 4. Gatekeeper parse failure after retry (D-02) → terminationReason='parse_failure'
 * 5. Unrecoverable error → terminationReason='error'
 */
export async function* runArenaOrchestration(
  params: OrchestrationParams
): AsyncGenerator<ArenaSSEEvent> {
  // 1. 加载关卡配置
  const loaded = loadLevel(params.worldId, params.levelId);
  if (!loaded) {
    yield { type: 'run_error', data: { error: `Level not found: ${params.worldId}/${params.levelId}` } };
    return;
  }
  const level = loaded.config;

  // 2. 解析 provider/model（per INTG-02 三角色独立配置）
  const gkProviderId = level.roleConfig?.gatekeeper?.providerId || params.defaultProviderId || '';
  const gkModel = level.roleConfig?.gatekeeper?.model || params.defaultModel || '';
  const chProviderId = level.roleConfig?.challenger?.providerId || params.defaultProviderId || '';
  const chModel = level.roleConfig?.challenger?.model || params.defaultModel || '';
  const grProviderId = level.roleConfig?.grader?.providerId || params.defaultProviderId || '';
  const grModel = level.roleConfig?.grader?.model || params.defaultModel || '';

  // 3. 创建 run 记录（DB-first）
  const run = createArenaRun({
    levelId: params.levelId,
    worldId: params.worldId,
    gatekeeperProviderId: gkProviderId,
    gatekeeperModel: gkModel,
    challengerProviderId: chProviderId,
    challengerModel: chModel,
    graderProviderId: grProviderId,
    graderModel: grModel,
  });

  yield { type: 'run_started', data: { runId: run.id, levelId: level.id, levelName: level.name } };

  // 4. 初始化 token tracker（per D-03 默认 200,000，关卡可覆盖）
  const tokenTracker = new TokenTracker(level.maxTokens);

  // 5. 编排循环（per D-01 synchronous while-loop）
  let turn = 0;
  let terminationReason: TerminationReason = 'max_turns'; // 默认
  const transcript: Array<{ role: 'user' | 'assistant'; content: string }> = [];
  const gradingTranscript: Array<{ role: string; content: string; turn: number }> = [];

  try {
    while (turn < level.maxTurns) {
      // 检查 token 预算
      if (!tokenTracker.hasEnoughForNextTurn()) {
        terminationReason = 'token_budget';
        break;
      }

      // --- Gatekeeper ---
      const gkResult = await callGatekeeper({
        transcript,
        level,
        abortSignal: params.abortSignal,
      });

      if (!gkResult) {
        // per D-02: 解析两次失败，终止
        terminationReason = 'parse_failure';
        // DB-first: 记录错误
        saveArenaMessage({ runId: run.id, role: 'system', content: 'Gatekeeper structured output parse failure after 2 attempts', turnNumber: turn });
        yield { type: 'run_error', data: { error: 'Gatekeeper output parse failure', turn } };
        break;
      }

      tokenTracker.consume(gkResult.usage.totalTokens);

      // DB-first: 存 Gatekeeper 消息
      saveArenaMessage({
        runId: run.id,
        role: 'gatekeeper',
        content: gkResult.output.message,
        turnNumber: turn,
        tokenUsage: { inputTokens: gkResult.usage.promptTokens, outputTokens: gkResult.usage.completionTokens },
      });

      yield { type: 'gatekeeper_message', data: { content: gkResult.output.message, turn } };
      yield { type: 'token_usage', data: { totalUsed: tokenTracker.totalUsed, remaining: tokenTracker.remaining } };

      // 添加到 transcript（Gatekeeper 消息作为 user 角色给 Challenger）
      transcript.push({ role: 'user', content: gkResult.output.message });
      gradingTranscript.push({ role: 'gatekeeper', content: gkResult.output.message, turn });

      // 检查 shouldEnd
      if (gkResult.output.shouldEnd) {
        terminationReason = 'gatekeeper_end';
        break;
      }

      // 检查 token 预算
      if (!tokenTracker.hasEnoughForNextTurn()) {
        terminationReason = 'token_budget';
        break;
      }

      // --- Challenger ---
      const challengerResult = await callChallenger({
        transcript,
        level,
        abortSignal: params.abortSignal,
        onDelta: (delta) => {
          // 注意：onDelta 是同步回调，但我们在 async generator 内。
          // SSE delta 事件需要通过 orchestrator 的 yield 机制发送。
          // 实现时可能需要使用事件队列或 callback → queue 模式。
        },
      });

      tokenTracker.consume(challengerResult.usage.totalTokens);

      // DB-first: 存 Challenger 完整消息
      saveArenaMessage({
        runId: run.id,
        role: 'challenger',
        content: challengerResult.fullContent,
        turnNumber: turn,
        tokenUsage: { inputTokens: challengerResult.usage.promptTokens, outputTokens: challengerResult.usage.completionTokens },
      });

      yield { type: 'challenger_message', data: { content: challengerResult.fullContent, turn } };
      yield { type: 'turn_completed', data: { turn, tokensUsed: tokenTracker.totalUsed } };
      yield { type: 'token_usage', data: { totalUsed: tokenTracker.totalUsed, remaining: tokenTracker.remaining } };

      // 添加到 transcript（Challenger 消息作为 assistant 角色）
      transcript.push({ role: 'assistant', content: challengerResult.fullContent });
      gradingTranscript.push({ role: 'challenger', content: challengerResult.fullContent, turn });

      turn++;
    }

    // 6. Grader 评分（per GRAD-01 对话结束后一次性评分）
    yield { type: 'grading_started', data: { turn } };

    const gradeResult = await callGrader({
      transcript: gradingTranscript,
      level,
      abortSignal: params.abortSignal,
    });

    tokenTracker.consume(gradeResult.usage.totalTokens);

    // DB-first: 存评分
    saveArenaGrade({
      runId: run.id,
      passed: gradeResult.output.passed,
      gradeData: gradeResult.output as unknown as Record<string, unknown>,
      tokenUsage: { inputTokens: gradeResult.usage.promptTokens, outputTokens: gradeResult.usage.completionTokens },
    });

    yield { type: 'grade_result', data: gradeResult.output };

    // 7. 更新 run 状态
    updateArenaRun(run.id, {
      status: 'completed',
      passed: gradeResult.output.passed ? 1 : 0,
      turnCount: turn,
      tokenUsageTotal: tokenTracker.totalUsed,
      terminationReason,
      completedAt: new Date().toISOString(),
    });

    yield { type: 'run_completed', data: { runId: run.id, passed: gradeResult.output.passed, terminationReason, turnCount: turn, totalTokens: tokenTracker.totalUsed } };

  } catch (err) {
    // 错误处理
    updateArenaRun(run.id, {
      status: 'failed',
      turnCount: turn,
      tokenUsageTotal: tokenTracker.totalUsed,
      terminationReason: 'error',
      completedAt: new Date().toISOString(),
    });

    yield { type: 'run_error', data: { error: err instanceof Error ? err.message : String(err), turn } };
    yield { type: 'run_completed', data: { runId: run.id, passed: null, terminationReason: 'error', turnCount: turn, totalTokens: tokenTracker.totalUsed } };
  }
}
```

**Challenger delta 流式推送设计说明：** 由于 `runArenaOrchestration` 是 async generator，而 Challenger 的 `onDelta` 是同步回调，无法直接在回调中 yield。实现时有两种方案：
- 方案 A：让 callChallenger 也返回 async generator（`for await (const chunk of callChallenger(...))`），在循环内逐 chunk yield SSE delta 事件
- 方案 B：在 orchestrator 的上层（API route）直接监听 Challenger stream，在 ReadableStream 的 start() 中处理

推荐方案 A，在 orchestrator 内使用 `for await` 消费 Challenger 的 async generator：
```typescript
for await (const delta of callChallengerStream(...)) {
  yield { type: 'challenger_delta', data: { delta, turn } };
}
```
