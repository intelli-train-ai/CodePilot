2. 创建 `src/arena/engine/gatekeeper.ts`（per ORCH-02 结构化输出、per D-02 解析失败重试一次）：

```typescript
import { generateText, Output } from 'ai';
import { buildAiModel } from './model-builder';
import { GatekeeperOutputSchema } from '../schemas/gatekeeper-output';
import type { GatekeeperOutput } from '../schemas/gatekeeper-output';
import type { LevelConfig } from '../schemas/level-config';

export interface GatekeeperCallResult {
  output: GatekeeperOutput;
  usage: { totalTokens: number; promptTokens: number; completionTokens: number };
}

/**
 * 调用 Gatekeeper LLM，获取结构化输出（含 shouldEnd）。
 * 使用 generateText + Output.object()（非 streamText），因为需要完整解析 shouldEnd 后才能决定是否继续循环。
 *
 * per D-02: 解析失败重试一次，再失败返回 null 表示 parse_failure。
 *
 * usage 获取方式：generateText 同步返回 result.usage，包含 promptTokens + completionTokens + totalTokens。
 * 这是 Gatekeeper 和 Grader 的 usage 来源（直接从 generateText result 读取）。
 */
export async function callGatekeeper(opts: {
  transcript: Array<{ role: 'user' | 'assistant'; content: string }>;
  level: LevelConfig;
  abortSignal?: AbortSignal;
}): Promise<GatekeeperCallResult | null> {
  const model = buildAiModel(
    opts.level.roleConfig?.gatekeeper?.providerId || '',
    opts.level.roleConfig?.gatekeeper?.model || '',
  );

  // 重试逻辑：per D-02，最多尝试 2 次
  for (let attempt = 0; attempt < 2; attempt++) {
    try {
      const { output, usage } = await generateText({
        model,
        output: Output.object({ schema: GatekeeperOutputSchema }),
        system: opts.level.gatekeeperSystemPrompt,
        messages: opts.transcript,
        abortSignal: opts.abortSignal || AbortSignal.timeout(60_000),
      });

      if (!output) {
        // Output.object() 返回 null 表示解析失败
        if (attempt === 0) continue; // 重试一次
        return null;
      }

      return {
        output,
        usage: {
          totalTokens: usage.totalTokens,
          promptTokens: usage.promptTokens,
          completionTokens: usage.completionTokens,
        },
      };
    } catch (err) {
      if (attempt === 0) continue; // 重试一次
      console.warn('[arena:gatekeeper] Structured output failed after 2 attempts:', err);
      return null;
    }
  }
  return null;
}
```

3. 创建 `src/arena/engine/challenger.ts`（per ORCH-01 Challenger 流式回复）。

**关键设计：callChallenger 是 async generator**，yield 每个 text delta，最终 return ChallengerResult（含 fullContent + usage）。编排器通过 `for await (const delta of callChallenger(...))` 消费 delta 并实时推 SSE，generator 结束后通过 return value 获取完整结果。

**usage 获取方式：** 不使用 `streamTextFromProvider`（它只 yield text 且不暴露 usage），而是直接调用 AI SDK `streamText`，通过 `buildAiModel` 构建 model。`streamText` 返回的 result 对象提供 `result.usage` (Promise)，在流结束后 resolve，包含 `promptTokens` + `completionTokens` + `totalTokens`。

```typescript
import { streamText } from 'ai';
import { buildAiModel } from './model-builder';
import type { LevelConfig } from '../schemas/level-config';

/** 每个 delta yield 的类型 */
export interface ChallengerDelta {
  text: string;
}

/** generator 结束后的完整结果 */
export interface ChallengerResult {
  fullContent: string;
  usage: { totalTokens: number; promptTokens: number; completionTokens: number };
}

/**
 * 调用 Challenger LLM，流式生成回复。
 *
 * 返回 async generator：
 * - yield ChallengerDelta（实时 text chunk，用于 SSE challenger_delta 事件推送）
 * - return ChallengerResult（流结束后，fullContent + 真实 usage 统计）
 *
 * 直接调用 AI SDK streamText（而非 streamTextFromProvider），因为需要：
 * 1. 访问 result.usage 获取真实 token 用量（streamTextFromProvider 不暴露 usage）
 * 2. 使用 messages 参数传入对话历史
 *
 * 仍通过 buildAiModel 走统一 provider 解析路径（per INTG-01）。
 */
export async function* callChallenger(opts: {
  transcript: Array<{ role: 'user' | 'assistant'; content: string }>;
  level: LevelConfig;
  abortSignal?: AbortSignal;
}): AsyncGenerator<ChallengerDelta, ChallengerResult> {
  const model = buildAiModel(
    opts.level.roleConfig?.challenger?.providerId || '',
    opts.level.roleConfig?.challenger?.model || '',
  );

  const result = streamText({
    model,
    system: opts.level.challengerSystemPrompt,
    messages: opts.transcript,
    maxOutputTokens: 4096,
    abortSignal: opts.abortSignal || AbortSignal.timeout(120_000),
  });

  const chunks: string[] = [];
  for await (const chunk of result.textStream) {
    chunks.push(chunk);
    yield { text: chunk };
  }

  // 流结束后获取真实 usage（result.usage 是 Promise，此时已 resolved）
  const usage = await result.usage;
  const fullContent = chunks.join('');

  return {
    fullContent,
    usage: {
      totalTokens: usage.totalTokens,
      promptTokens: usage.promptTokens,
      completionTokens: usage.completionTokens,
    },
  };
}
```

4. 创建 `src/arena/engine/grader.ts`（per GRAD-01 对话结束后一次性评分、per GRAD-04 Zod 校验、per GRAD-05 改进建议引用轮次）：

```typescript
import { generateText, Output } from 'ai';
import { buildAiModel } from './model-builder';
import { GraderOutputSchema } from '../schemas/grader-output';
import type { GraderOutput } from '../schemas/grader-output';
import type { LevelConfig, RubricItem } from '../schemas/level-config';

/**
 * 调用 Grader LLM，一次性评分完整对话 transcript。
 * 使用 generateText + Output.object() + GraderOutputSchema 获取 Zod 校验过的结构化输出。
 *
 * usage 获取方式：与 Gatekeeper 相同，直接从 generateText result.usage 读取。
 */
export async function callGrader(opts: {
  transcript: Array<{ role: string; content: string; turn: number }>;
  level: LevelConfig;
  abortSignal?: AbortSignal;
}): Promise<{ output: GraderOutput; usage: { totalTokens: number; promptTokens: number; completionTokens: number } }> {
  const graderPrompt = buildGraderPrompt(opts.transcript, opts.level.rubric);

  const defaultGraderSystem = `You are a grading judge for an AI agent evaluation. You will receive a conversation transcript between a Gatekeeper (tester) and a Challenger (AI agent being tested), along with a grading rubric. Evaluate the Challenger's performance strictly according to the rubric.

For required criteria: determine Pass or Fail with a clear reason.
For performance dimensions: assign a grade (A/B/C/D) based on the rubric's grade descriptions, with a clear reason.
Provide up to 3 specific improvement suggestions, each referencing a conversation turn number.
The overall "passed" field should be true ONLY if ALL required criteria passed.`;

  const systemPrompt = opts.level.graderSystemPrompt || defaultGraderSystem;

  const model = buildAiModel(
    opts.level.roleConfig?.grader?.providerId || '',
    opts.level.roleConfig?.grader?.model || '',
  );

  const { output, usage } = await generateText({
    model,
    output: Output.object({ schema: GraderOutputSchema }),
    system: systemPrompt,
    prompt: graderPrompt,
    abortSignal: opts.abortSignal || AbortSignal.timeout(120_000),
  });

  if (!output) {
    throw new Error('[arena:grader] Failed to generate structured grading output');
  }

  return {
    output,
    usage: {
      totalTokens: usage.totalTokens,
      promptTokens: usage.promptTokens,
      completionTokens: usage.completionTokens,
    },
  };
}

/**
 * 构建 Grader 的 prompt，包含 rubric 详情和完整对话 transcript。
 */
function buildGraderPrompt(
  transcript: Array<{ role: string; content: string; turn: number }>,
  rubric: RubricItem[],
): string {
  let prompt = '## Grading Rubric\n\n';

  for (const item of rubric) {
    prompt += `### ${item.name} (${item.type})\n`;
    prompt += `${item.description}\n`;
    if (item.type === 'performance' && item.gradeDescriptions) {
      for (const [grade, desc] of Object.entries(item.gradeDescriptions)) {
        prompt += `- **${grade}**: ${desc}\n`;
      }
    }
    prompt += '\n';
  }

  prompt += '## Conversation Transcript\n\n';
  for (const msg of transcript) {
    const roleLabel = msg.role === 'gatekeeper' ? 'Gatekeeper' : 'Challenger';
    prompt += `**Turn ${msg.turn} - ${roleLabel}:**\n${msg.content}\n\n`;
  }

  prompt += '## Instructions\n\n';
  prompt += 'Evaluate the Challenger based on the rubric above. For each required criterion, determine Pass or Fail. For each performance dimension, assign a grade (A/B/C/D) using the grade descriptions as anchors. Provide up to 3 improvement suggestions referencing specific turn numbers.';

  return prompt;
}
```

```typescript
const DEFAULT_TOKEN_BUDGET = 200_000;
const GRADER_RESERVE = 8_000; // 为 Grader 预留的 token 空间

export class TokenTracker {
  private budget: number;
  private used: number = 0;

  constructor(budget?: number) {
    this.budget = budget ?? DEFAULT_TOKEN_BUDGET;
  }

  /** 记录 token 消耗 */
  consume(tokens: number): void {
    this.used += tokens;
  }

  /** 剩余可用 token */
  get remaining(): number {
    return this.budget - this.used;
  }

  /** 已消耗总量 */
  get totalUsed(): number {
    return this.used;
  }

  /** 预算是否已耗尽 */
  get isExhausted(): boolean {
    return this.remaining <= 0;
  }

  /** 是否有足够 token 继续（含 Grader 预留） */
  hasEnoughForNextTurn(): boolean {
    return this.remaining > GRADER_RESERVE;
  }

  /** 是否有足够 token 进行指定操作 */
  hasEnoughFor(estimatedTokens: number): boolean {
    return this.remaining >= estimatedTokens;
  }
}
```

**与 Task 1 的接口协调：callChallenger 是 async generator**，编排器使用 `for await` 消费：

```typescript
import { callGatekeeper } from './gatekeeper';
import { callChallenger } from './challenger';
import { callGrader } from './grader';
import { TokenTracker } from './token-tracker';
import { createArenaRun, updateArenaRun, saveArenaMessage, saveArenaGrade } from '../db';
import { loadLevel } from '../level-loader';
import type { ArenaSSEEvent, TerminationReason } from '../types';

export interface OrchestrationParams {
  worldId: string;
  levelId: string;
  defaultProviderId?: string;
  defaultModel?: string;
  abortSignal?: AbortSignal;
}

export interface OrchestrationResult {
  runId: string;
  passed: boolean | null;
  terminationReason: TerminationReason;
  turnCount: number;
  totalTokens: number;
}

/**
 * 核心编排循环。per D-01：单 API 调用驱动整个 run。
 *
 * 返回 async generator，yield ArenaSSEEvent 序列。
 *
 * SSE 事件的 DB-first 规则：
 * - challenger_delta：实时 yield（无 DB 写入，delta 是临时流数据）
 * - challenger_message：saveArenaMessage 入库后才 yield
 * - gatekeeper_message：saveArenaMessage 入库后才 yield
 * - grade_result：saveArenaGrade 入库后才 yield
 *
 * 循环终止条件（按优先级）：
 * 1. Gatekeeper shouldEnd=true (ORCH-02)
 * 2. Token budget exhausted (ORCH-04, D-03) -- 使用真实 usage 值
 * 3. maxTurns reached (ORCH-03)
 * 4. Gatekeeper parse failure after retry (D-02)
 * 5. Unrecoverable error
 */
export async function* runArenaOrchestration(
  params: OrchestrationParams
): AsyncGenerator<ArenaSSEEvent> {
  // 1. 加载关卡配置
  const loaded = loadLevel(params.worldId, params.levelId);
  if (!loaded) {
    yield { type: 'run_error', data: { error: `Level not found: ${params.worldId}/${params.levelId}` } };
    return;
  }
  const level = loaded.config;

  // 2. 创建 run 记录（DB-first）
  const run = createArenaRun({
    levelId: params.levelId,
    worldId: params.worldId,
    gatekeeperProviderId: level.roleConfig?.gatekeeper?.providerId || params.defaultProviderId || '',
    gatekeeperModel: level.roleConfig?.gatekeeper?.model || params.defaultModel || '',
    challengerProviderId: level.roleConfig?.challenger?.providerId || params.defaultProviderId || '',
    challengerModel: level.roleConfig?.challenger?.model || params.defaultModel || '',
    graderProviderId: level.roleConfig?.grader?.providerId || params.defaultProviderId || '',
    graderModel: level.roleConfig?.grader?.model || params.defaultModel || '',
  });

  yield { type: 'run_started', data: { runId: run.id, levelId: level.id, levelName: level.name } };

  // 3. 初始化 token tracker（per D-03 默认 200,000，关卡可覆盖）
  const tokenTracker = new TokenTracker(level.maxTokens);

  // 4. 编排循环（per D-01 synchronous while-loop）
  let turn = 0;
  let terminationReason: TerminationReason = 'max_turns'; // 默认
  const transcript: Array<{ role: 'user' | 'assistant'; content: string }> = [];
  const gradingTranscript: Array<{ role: string; content: string; turn: number }> = [];

  try {
    while (turn < level.maxTurns) {
      // 检查 token 预算
      if (!tokenTracker.hasEnoughForNextTurn()) {
        terminationReason = 'token_budget';
        break;
      }

      // --- Gatekeeper ---
      const gkResult = await callGatekeeper({
        transcript,
        level,
        abortSignal: params.abortSignal,
      });

      if (!gkResult) {
        terminationReason = 'parse_failure';
        saveArenaMessage({ runId: run.id, role: 'system', content: 'Gatekeeper structured output parse failure after 2 attempts', turnNumber: turn });
        yield { type: 'run_error', data: { error: 'Gatekeeper output parse failure', turn } };
        break;
      }

      // 真实 usage 来自 generateText result
      tokenTracker.consume(gkResult.usage.totalTokens);

      // DB-first: 存 Gatekeeper 消息后才 yield
      saveArenaMessage({
        runId: run.id,
        role: 'gatekeeper',
        content: gkResult.output.message,
        turnNumber: turn,
        tokenUsage: { inputTokens: gkResult.usage.promptTokens, outputTokens: gkResult.usage.completionTokens },
      });

      yield { type: 'gatekeeper_message', data: { content: gkResult.output.message, turn } };
      yield { type: 'token_usage', data: { totalUsed: tokenTracker.totalUsed, remaining: tokenTracker.remaining } };

      transcript.push({ role: 'user', content: gkResult.output.message });
      gradingTranscript.push({ role: 'gatekeeper', content: gkResult.output.message, turn });

      // 检查 shouldEnd
      if (gkResult.output.shouldEnd) {
        terminationReason = 'gatekeeper_end';
        break;
      }

      // 检查 token 预算
      if (!tokenTracker.hasEnoughForNextTurn()) {
        terminationReason = 'token_budget';
        break;
      }

      // --- Challenger (async generator) ---
      // callChallenger 是 async generator，yield ChallengerDelta，return ChallengerResult
      const challengerGen = callChallenger({
        transcript,
        level,
        abortSignal: params.abortSignal,
      });

      // 消费 delta 并实时推 SSE（无 DB 写入，delta 是临时流数据）
      let challengerResult;
      while (true) {
        const { value, done } = await challengerGen.next();
        if (done) {
          // done=true 时 value 是 return 值（ChallengerResult）
          challengerResult = value;
          break;
        }
        // done=false 时 value 是 yield 值（ChallengerDelta）
        yield { type: 'challenger_delta', data: { delta: value.text, turn } };
      }

      // 真实 usage 来自 streamText result.usage（已在 generator return 中 await）
      tokenTracker.consume(challengerResult.usage.totalTokens);

      // DB-first: 存 Challenger 完整消息后才 yield challenger_message
      saveArenaMessage({
        runId: run.id,
        role: 'challenger',
        content: challengerResult.fullContent,
        turnNumber: turn,
        tokenUsage: { inputTokens: challengerResult.usage.promptTokens, outputTokens: challengerResult.usage.completionTokens },
      });

      yield { type: 'challenger_message', data: { content: challengerResult.fullContent, turn } };
      yield { type: 'turn_completed', data: { turn, tokensUsed: tokenTracker.totalUsed } };
      yield { type: 'token_usage', data: { totalUsed: tokenTracker.totalUsed, remaining: tokenTracker.remaining } };

      transcript.push({ role: 'assistant', content: challengerResult.fullContent });
      gradingTranscript.push({ role: 'challenger', content: challengerResult.fullContent, turn });

      turn++;
    }

    // 5. Grader 评分（per GRAD-01 对话结束后一次性评分）
    yield { type: 'grading_started', data: { turn } };

    const gradeResult = await callGrader({
      transcript: gradingTranscript,
      level,
      abortSignal: params.abortSignal,
    });

    tokenTracker.consume(gradeResult.usage.totalTokens);

    // DB-first: 存评分后才 yield
    saveArenaGrade({
      runId: run.id,
      passed: gradeResult.output.passed,
      gradeData: gradeResult.output as unknown as Record<string, unknown>,
      tokenUsage: { inputTokens: gradeResult.usage.promptTokens, outputTokens: gradeResult.usage.completionTokens },
    });

    yield { type: 'grade_result', data: gradeResult.output };

    // 6. 更新 run 状态
    updateArenaRun(run.id, {
      status: 'completed',
      passed: gradeResult.output.passed ? 1 : 0,
      turnCount: turn,
      tokenUsageTotal: tokenTracker.totalUsed,
      terminationReason,
      completedAt: new Date().toISOString(),
    });

    yield { type: 'run_completed', data: { runId: run.id, passed: gradeResult.output.passed, terminationReason, turnCount: turn, totalTokens: tokenTracker.totalUsed } };

  } catch (err) {
    updateArenaRun(run.id, {
      status: 'failed',
      turnCount: turn,
      tokenUsageTotal: tokenTracker.totalUsed,
      terminationReason: 'error',
      completedAt: new Date().toISOString(),
    });

    yield { type: 'run_error', data: { error: err instanceof Error ? err.message : String(err), turn } };
    yield { type: 'run_completed', data: { runId: run.id, passed: null, terminationReason: 'error', turnCount: turn, totalTokens: tokenTracker.totalUsed } };
  }
}
```
